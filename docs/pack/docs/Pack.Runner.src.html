14 | runCmd : HasIO io => EitherT PackErr io ()
15 | runCmd = do
16 |   (c,cmd) <- getConfig cmd PrintHelp loglevel
17 |   case cmd of
18 |     Completion a b     => env c >>= complete a b
19 |     CompletionScript f => putStrLn (completionScript f)
20 |     Query m s          => env c >>= query m s
21 |     Fuzzy m s          => idrisEnv c >>= fuzzy m s
22 |     UpdateDB           => updateDB c
23 |     Run (Right p) args => idrisEnv c >>= execApp p args
24 |     Run (Left p)  args => idrisEnv c >>= runIpkg p args
25 |     Repl p             => idrisEnv c >>= idrisRepl p
26 |     Build p            => idrisEnv c >>= build p
27 |     BuildDeps p        => idrisEnv c >>= buildDeps p
28 |     Typecheck p        => idrisEnv c >>= typecheck p
29 |     PrintHelp          => putStrLn usageInfo
30 |     Install ps         => idrisEnv c >>= \e => traverse_ (installLib e) ps
31 |     Remove ps          => idrisEnv c >>= \e => traverse_ (remove e) ps
32 |     InstallApp ps      => idrisEnv c >>= \e => traverse_ (installApp e) ps
33 |     PackagePath        => env c >>= putStrLn . packagePathDirs
34 |     LibsPath           => env c >>= putStrLn . packageLibDirs
35 |     DataPath           => env c >>= putStrLn . packageDataDirs
36 |     Info               => env c >>= printInfo
37 |     New dir pty p      => idrisEnv c >>= new dir pty p
38 |     Switch db          => case db == MkDBName "latest" of
39 |       True  => do
40 |         updateDB c
41 |         db  <- defaultColl c.packDir
42 |         env <- idrisEnv ({collection := db} c)

44 |         writeCollection env
