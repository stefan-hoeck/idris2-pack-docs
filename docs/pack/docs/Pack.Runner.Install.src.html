 55 | -- When linking to a binary from a package collection's
 56 | -- `bin` directory, we distinguish between applications,
 57 | -- which need acceess to the Idris package path and those,
 58 | -- which don't. For the former, we create a wrapper script
 59 | -- where we first set the `IDRIS2_PACKAGE_PATH` variable
 60 | -- before invoking the binary, for the latter we create just
 61 | -- a symlink.
 62 | appLink :  HasIO io
 63 |         => (exec     : File Abs)
 64 |         -> (target   : File Abs)
 65 |         -> (packPath : Maybe $ File Abs)
 66 |         -> EitherT PackErr io ()
 67 | appLink (MkF p b) target Nothing  = link (p /> b) target
 68 | appLink exec target (Just p)       =
 69 |   let content = """
 70 |       #!/bin/sh
 71 |
 72 |       export IDRIS2_PACKAGE_PATH="$(\{p} package-path)"
 73 |       export IDRIS2_LIBS="$(\{p} libs-path)"
 74 |       export IDRIS2_DATA="$(\{p} data-path)"
 75 |       "\{exec}" "$@"
 76 |       """
 77 |    in write target content >> sys "chmod +x \{target}"
