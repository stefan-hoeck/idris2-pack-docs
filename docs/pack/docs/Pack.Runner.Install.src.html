 27 | pthStr : PackDir => Bool -> String
 28 | pthStr False = ""
 29 | pthStr True = """
 30 |   export IDRIS2_PACKAGE_PATH="$(\{packExec} package-path)"
 31 |   export IDRIS2_LIBS="$(\{packExec} libs-path)"
 32 |   export IDRIS2_DATA="$(\{packExec} data-path)"
 33 |   """
 34 | -- When linking to a binary from pack's `bin` directory,
 35 | -- we distinguish between applications,
 36 | -- which need acceess to the Idris package path and those,
 37 | -- which don't. For the former, we create a wrapper script
 38 | -- where we first set the `IDRIS2_PACKAGE_PATH` variable
 39 | -- before invoking the binary. For both cases, we let pack
 40 | -- decide which version to use.
 41 | appLink :  HasIO io
 42 |         => PackDir
 43 |         => (exec        : Body)
 44 |         -> (app         : PkgName)
 45 |         -> (withPkgPath : Bool)
 46 |         -> EitherT PackErr io ()
 47 | appLink exec app withPkgPath =
 48 |   let target  := MkF packBinDir exec
 49 |       content := """
 50 |       #!/bin/sh
 51 |
 52 |       APPLICATION="$(\{packExec} app-path \{app})"
 53 |       \{pthStr withPkgPath}
 54 |
 55 |       $APPLICATION "$@"
 56 |       """
 57 |    in write target content >> sys "chmod +x \{target}"
