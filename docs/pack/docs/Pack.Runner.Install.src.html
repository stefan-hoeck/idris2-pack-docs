 49 | -- When linking to a binary from pack's `bin` directory,
 50 | -- we distinguish between applications,
 51 | -- which need acceess to the Idris package path and those,
 52 | -- which don't. For the former, we create a wrapper script
 53 | -- where we first set the `IDRIS2_PACKAGE_PATH` variable
 54 | -- before invoking the binary. For both cases, we let pack
 55 | -- decide which version to use.

 65 |   let
 66 |       interp  := case cg of

 70 |       content := """
 71 |       #!/bin/sh

 73 |       APPLICATION="$(\{packExec} app-path \{app})"
